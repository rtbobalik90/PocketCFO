/* ========================================================= Pocket CFO — debt.js Debt payoff simulation engine. Avalanche (highest APR first) vs Snowball (lowest balance first). Month-by-month simulation with interest accrual, payoff timeline, interest saved vs minimums-only, and an honesty check on the user's extra-payment plan. ========================================================= */ (function () { /** * Simulate payoff. * @param {Array} debts [{name, balance, apr, minPayment}] * @param {number} extra extra dollars per month on top of all minimums * @param {string} strategy 'avalanche' | 'snowball' | 'minimum' * @returns {months, totalInterest, totalPaid, order[], balanceSeries[], perDebt[], stalled} */ function simulate(debts, extra, strategy) { const ds = debts.map(d => ({ ...d, paidInterest: 0, payoffMonth: null })); const order = rank(ds, strategy); let months = 0, totalInterest = 0, totalPaid = 0; const balanceSeries = [ds.reduce((s, d) => s + d.balance, 0)]; const CAP = 600; // 50 years — beyond this, the plan doesn't work let stalled = false; while (ds.some(d => d.balance > 0.005) && months < CAP) { months++; let pool = extra; // accrue interest + pay minimums for (const d of ds) { if (d.balance <= 0.005) continue; const interest = d.balance * (d.apr / 100) / 12; d.balance += interest; d.paidInterest += interest; totalInterest += interest; const pay = Math.min(d.balance, d.minPayment); d.balance -= pay; totalPaid += pay; if (d.minPayment <= interest) stalled = true; // minimum doesn't cover interest } // extra to target order if (strategy !== 'minimum') { for (const name of order) { if (pool <= 0) break; const d = ds.find(x => x.name === name && x.balance > 0.005); if (!d) continue; const pay = Math.min(d.balance, pool); d.balance -= pay; pool -= pay; totalPaid += pay; } } for (const d of ds) if (d.balance <= 0.005 && d.payoffMonth === null) d.payoffMonth = months; balanceSeries.push(ds.reduce((s, d) => s + Math.max(0, d.balance), 0)); } return { months, totalInterest, totalPaid, order, balanceSeries, perDebt: ds.map(d => ({ name: d.name, payoffMonth: d.payoffMonth, paidInterest: d.paidInterest })), finished: months < CAP, stalled }; } function rank(debts, strategy) { const c = [...debts]; if (strategy === 'snowball') c.sort((a, b) => a.balance - b.balance); else c.sort((a, b) => b.apr - a.apr); // avalanche default return c.map(d => d.name); } /** * Full analysis: chosen strategy vs the other vs minimums-only, * plus a plain-English honesty check against safe-to-save capacity. */ function analyze(debts, extra, strategy, safeMonthly) { if (!debts.length) return null; const chosen = simulate(debts, extra, strategy); const alt = simulate(debts, extra, strategy === 'avalanche' ? 'snowball' : 'avalanche'); const minOnly = simulate(debts, 0, 'minimum'); const advice = []; const fmt = window.Store.fmt; if (!minOnly.finished || minOnly.stalled) { advice.push({ level: 'bad', text: 'On minimum payments alone, at least one of these debts barely moves — interest eats the payment. Extra payments are not optional here; they are the plan.' }); } if (chosen.finished) { const saved = minOnly.totalInterest - chosen.totalInterest; if (saved > 0 && minOnly.finished) advice.push({ level: 'good', text: `This plan saves ${fmt(saved)} in interest and finishes ${minOnly.months - chosen.months} months sooner than minimums only.` }); } else { advice.push({ level: 'bad', text: 'This plan never finishes within 50 years. Increase the extra payment or restructure the debt.' }); } if (strategy === 'snowball' && alt.finished && chosen.finished && alt.totalInterest < chosen.totalInterest - 25) { advice.push({ level: 'info', text: `Snowball costs ${fmt(chosen.totalInterest - alt.totalInterest)} more than avalanche here. If the quick wins keep you going, that's money well spent — but know the price.` }); } if (strategy === 'avalanche' && chosen.finished) { advice.push({ level: 'info', text: 'Avalanche is the mathematically cheapest path. If motivation dips, switching to snowball for one small debt is fine.' }); } if (typeof safeMonthly === 'number') { if (extra > safeMonthly * 1.25 && safeMonthly >= 0) { advice.push({ level: 'warn', text: `Honest check: ${fmt(extra)}/mo extra is above what your last 3 months of cash flow support (~${fmt(safeMonthly)}). Plans that require a perfect month fail on the first bad one. Set the extra to something you can hit even in a rough month.` }); } else if (extra < safeMonthly * 0.4 && safeMonthly > 100 && chosen.months > 12) { advice.push({ level: 'warn', text: `You have roughly ${fmt(safeMonthly)}/mo of real capacity and you're only sending ${fmt(extra)}. At these APRs, comfort is expensive. Consider raising the extra payment.` }); } } const highApr = debts.filter(d => d.apr >= 20); if (highApr.length) advice.push({ level: 'info', text: `${highApr.map(d => d.name).join(' and ')} ${highApr.length > 1 ? 'are' : 'is'} above 20% APR. Nothing you invest or save will reliably beat paying ${highApr.length > 1 ? 'these' : 'this'} off.` }); return { chosen, alt, minOnly, advice }; } window.DebtEngine = { simulate, analyze }; })();
